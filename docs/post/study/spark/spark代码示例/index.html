SparkSQL相关

from pyspark.sql import SparkSession
spark = SparkSession.builder.appName("sparkSQL").master("local[*]").config().getOrCreate()
sc = spark.sparkContext()

数据输入输出

RDD数据输入输出

# 输入
rdd = sc.parallelize([1,2,3]) # 从内存创建
rdd = sc.textFile(path="",minPartitions) # 以行为单位读取文件
rdd = sc.wholeTextFiles() # 读取整个文件

# 输出
rdd.saveAsTextFile(path)
rdd.saveAsObjectFile(path)
rdd.saveAsSequenceFile(path) # 要求必须是key-value RDD

DataFrame数据输入输出

# 输入
# 直接创建
df = spark.createDataFrame(Optional[rdd, pandas, data])

# 从文件中读取
df = spark.read.format ("...") [.option("…")].load("…")
df = spark.read.format("csv").\
    option("sep", ":").\
    option("header", False).\
    option("encoding", "utf-8").\
    schema("name STRING, age INT, job STRING").\
    load(path)
# format支持：csv, jdbc, json, orc, parquet, textFile
# option 表示传入format选择的方法中的相应参数。

# 从RDD转化
df = rdd.toDF()
rdd = df.rdd()

# 输出
df.write.mode().format().option(K, V).save(PATH)
df.select(F.concat_ws("---", "user_id")).write.mode("overwrite").option("header", True).save(path)
# mode 表示设定模式，比如append、overwrite等。

SQL UDF语法支持

python只支持UDF(user-defined-function)，表示一对一关系

sparksession.udf.register(udf_name, udf_function, return_type)
# 使用：
sparksession.udf.regisiter("udf1", num_add_1, IntergerType())
spark.sql.selectExpr("udf1(df['nums']"))。 或者df.udf1(df["nums"])

Spark MLlib

from pyspark.ml.evaluation import BinaryClassificationEvaluator

va = VectorAssembler().setInputCols(["int1", "int2", "int3"]).setOutputCols("features")

ss = Standard().setInputCol("features")
lr = LogisticRegression().setLabelCol("label").setFeaturesCol("features")
stages = [ss, lr]
pipeline = Pipeline().setStages(stages)

evaluator = BinaryClassificationEvaluator().setMetricName("areaUnderROC").setRawPredictionCol("prediction").setLabelCol("label")

tvs = TrainValidationSplit().setTrainRatio(0.75).setEstimatorParamMaps(params).setEstimator(pipeline).setEvaluator(evaluator)

tvsFitted = tvs.fit(train)

evaluator.evaluate(tvsFitted.transform(test))
