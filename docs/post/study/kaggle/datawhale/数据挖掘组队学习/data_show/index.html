def getres1(row):
    return len([x for x in row.values if type(x)==int and x==-1])

def getresneg(row):
    return len([x for x in row.values if type(x) in [int, float] and x<0])

from sklearn.preprocessing import StandardScaler
#参数说明：dispersed为True代表是离散数据，则采用Knn分类器，为False代表是连续数据，则采用Knn回归器
def knn_missing_filled(data, train_col, test_col, n_neighbors=5, weights="uniform"):
    # 对train_col获取特征，然后min_max归一化, 
    scaler = StandardScaler()
    data[data[test_col]<0] = np.nan
    train_data = scaler.fit_transform(data[train_col])
    #train_data = data[train_col].apply(lambda x:(x - np.min(x)) / (np.max(x) - np.min(x)))
    test_data = data[test_col].values
    array = np.concatenate([train_data, test_data], axis=1)
    imputer = KNNImputer(n_neighbors=n_neighbors, weights=weights)
    imputed = imputer.fit_transform(array)
    df_imputered = pd.DataFrame(imputed[:, len(train_col):], columns=test_col)
    return df_imputered
